00032 /****************************************************************************
00033  * ResumeGui
00034  *
00035  * Signals the GUI thread to start, and resumes the thread. This is called
00036  * after finishing the removal/insertion of new elements, and after initial
00037  * GUI setup.
00038  ***************************************************************************/
00039 //static 
00040 void ResumeGui()
00041 {
00042         guiHalt = false;
00043         LWP_ResumeThread (guithread);
00044 }
00045 
00046 /****************************************************************************
00047  * HaltGui
00048  *
00049  * Signals the GUI thread to stop, and waits for GUI thread to stop
00050  * This is necessary whenever removing/inserting new elements into the GUI.
00051  * This eliminates the possibility that the GUI is in the middle of accessing
00052  * an element that is being changed.
00053  ***************************************************************************/
00054 //static 
00055 void HaltGui()
00056 {
00057         guiHalt = true;
00058 
00059         // wait for thread to finish
00060         while(!LWP_ThreadIsSuspended(guithread))
00061                 usleep(THREAD_SLEEP);
00062 }
